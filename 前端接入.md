### 3. API 服务集成

#### 创建工作流 API 客户端

```typescript
// api/workflowClient.ts
import type { 
  WorkflowDefinition, 
  ExecuteWorkflowRequest,
  ExecuteWorkflowResponse 
} from 'workflow-engine-frontend/src/types/workflow';

class WorkflowClient {
  private baseUrl: string;
  private headers: Record<string, string>;

  constructor(baseUrl: string, apiKey?: string) {
    this.baseUrl = baseUrl.replace(/\/$/, '');
    this.headers = {
      'Content-Type': 'application/json',
      ...(apiKey && { 'Authorization': `Bearer ${apiKey}` })
    };
  }

  // 获取工作流列表
  async getWorkflows(): Promise<WorkflowDefinition[]> {
    const response = await fetch(`${this.baseUrl}/api/workflows`, {
      headers: this.headers
    });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return response.json();
  }

  // 执行工作流
  async executeWorkflow(workflow: WorkflowDefinition): Promise<ExecuteWorkflowResponse> {
    const request: ExecuteWorkflowRequest = { workflow };
    const response = await fetch(`${this.baseUrl}/api/workflows/execute`, {
      method: 'POST',
      headers: this.headers,
      body: JSON.stringify(request)
    });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return response.json();
  }

  // 创建工作流
  async createWorkflow(workflow: WorkflowDefinition): Promise<{ id: string }> {
    const response = await fetch(`${this.baseUrl}/api/workflows`, {
      method: 'POST',
      headers: this.headers,
      body: JSON.stringify(workflow)
    });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return response.json();
  }
}

export const workflowClient = new WorkflowClient(
  process.env.REACT_APP_API_URL || 'http://localhost:8000',
  process.env.REACT_APP_API_KEY
);
```

```tsx
// components/CustomNode.tsx
import React from 'react';
import { Handle, Position, type NodeProps } from '@xyflow/react';

interface CustomNodeData {
  label: string;
  type: string;
  status: 'pending' | 'running' | 'success' | 'error';
  customData?: any;
}

export const CustomNode: React.FC<NodeProps<CustomNodeData>> = ({ data, selected }) => {
  const statusColors = {
    pending: 'border-gray-300 bg-white',
    running: 'border-blue-500 bg-blue-50',
    success: 'border-green-500 bg-green-50',
    error: 'border-red-500 bg-red-50'
  };

  return (
    <div className={`rounded-lg border-2 p-4 shadow-md ${statusColors[data.status]} ${selected ? 'ring-2 ring-blue-400' : ''}`}>
      <Handle type="target" position={Position.Top} className="w-3 h-3" />
      <div className="text-sm font-semibold">{data.label}</div>
      <div className="text-xs text-gray-600 mt-1">{data.type}</div>
      {data.customData && (
        <div className="text-xs mt-2 p-2 bg-gray-100 rounded">
          {JSON.stringify(data.customData, null, 2)}
        </div>
      )}
      <Handle type="source" position={Position.Bottom} className="w-3 h-3" />
    </div>
  );
};
```

## 📋 高级集成方案

### 方案一：完整工作流系统集成

```tsx
// IntegratedWorkflowSystem.tsx
import React, { useCallback, useState } from 'react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { ReactFlowProvider } from '@xyflow/react';
import { 
  TopBar, 
  CanvasPanel, 
  ChatPanel, 
  RightDrawer, 
  BottomPanel 
} from 'workflow-engine-frontend/src/components/layout';
import { useWorkflowStore } from 'workflow-engine-frontend/src/store/workflowStore';
import { workflowClient } from './api/workflowClient';
import './workflow-styles.css';

const queryClient = new QueryClient();

export const IntegratedWorkflowSystem: React.FC = () => {
  const [drawerOpen, setDrawerOpen] = useState(false);
  const workflow = useWorkflowStore(state => state.workflow);
  const nodeStatusMap = useWorkflowStore(state => state.nodeStatusMap);
  const selectedNodeId = useWorkflowStore(state => state.selectedNodeId);
  const setSelectedNodeId = useWorkflowStore(state => state.setSelectedNodeId);
  const setWorkflow = useWorkflowStore(state => state.setWorkflow);

  const handleWorkflowChange = useCallback((newWorkflow: any) => {
    setWorkflow(newWorkflow, true);
  }, [setWorkflow]);

  const handleExecute = useCallback(async () => {
    if (!workflow) return;
    try {
      const result = await workflowClient.executeWorkflow(workflow);
      console.log('执行结果:', result);
    } catch (error) {
      console.error('执行失败:', error);
    }
  }, [workflow]);

  return (
    <QueryClientProvider client={queryClient}>
      <ReactFlowProvider>
        <div className="flex flex-col h-screen bg-gray-50">
          <TopBar
            workflowName={workflow?.name || '新工作流'}
            versionStatus="unsaved"
            isSaving={false}
            isExecuting={false}
            onWorkflowNameChange={(name) => {
              if (workflow) setWorkflow({ ...workflow, name }, true);
            }}
            onSave={() => console.log('保存工作流')}
            onExecute={handleExecute}
            onStop={() => console.log('停止执行')}
            onImportWorkflow={setWorkflow}
            onExportWorkflow={() => console.log('导出工作流')}
          />
          
          <div className="flex flex-1 overflow-hidden">
            <div className="flex-1 flex">
              <CanvasPanel
                workflow={workflow}
                nodeStatusMap={nodeStatusMap}
                selectedNodeId={selectedNodeId}
                onSelectNode={setSelectedNodeId}
                onWorkflowChange={handleWorkflowChange}
              />
              <RightDrawer
                open={drawerOpen}
                onOpenChange={setDrawerOpen}
                node={workflow?.nodes.find(n => n.id === selectedNodeId)}
                onNodeUpdate={(nodeId, updates) => {
                  // 更新节点逻辑
                }}
              />
            </div>
            <ChatPanel
              messages={[]}
              onSendMessage={(message) => console.log('发送消息:', message)}
              disabled={false}
            />
          </div>
          
          <BottomPanel
            execution={null}
            stats={null}
          />
        </div>
      </ReactFlowProvider>
    </QueryClientProvider>
  );
};
```

## 🔌 后端 API 规范

### 工作流数据结构

```typescript
interface WorkflowDefinition {
  name: string;
  description?: string;
  nodes: WorkflowNode[];
  edges: WorkflowEdge[];
  variables?: Record<string, any>;
}

interface WorkflowNode {
  id: string;
  type: string;
  config?: {
    title?: string;
    description?: string;
    params?: Record<string, any>;
    [key: string]: any;
  };
  position?: {
    x: number;
    y: number;
  };
}

interface WorkflowEdge {
  source: string;
  target: string;
  condition?: string;
  branch?: string;
}
```

### 必需 API 端点

```typescript
// 工作流管理
GET    /api/workflows              // 获取工作流列表

PUT    /api/workflows/:id          // 更新工作流
DELETE /api/workflows/:id          // 删除工作流

// 工作流执行
```

## 🔧 常见问题解决

### Q: 集成后样式冲突怎么办？
A: 使用 CSS Modules 或 styled-components 进行样式隔离，或者添加特定的命名空间前缀。

### Q: 如何处理不同的后端 API 格式？
A: 在 API 客户端层添加适配器，将后端格式转换为前端标准格式。

### Q: 如何自定义节点类型？
A: 创建自定义节点组件并在 nodeTypes 配置中注册，然后通过工作流定义中的 type 字段使用。

### Q: 集成后性能问题如何优化？
A: 检查是否有过度的重渲染，使用 React DevTools Profiler 分析性能瓶颈，考虑使用虚拟化技术处理大量节点。
